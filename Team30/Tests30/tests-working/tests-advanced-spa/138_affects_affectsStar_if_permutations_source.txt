procedure permOne {
    read x;
    read y;
    read z;
    if (x > 0) {
		x = y + z;
		if (y > 0){
			x = x + 1;
			if (z > 0) {
				z = z - y;
			} else {
				z = z - y;
			}
			y = y + 1;
		} else { 
			y = y + 1;
		}
		y = x + z;
    } else { 
		y = x + z;
	}
	x = x + z;
}
